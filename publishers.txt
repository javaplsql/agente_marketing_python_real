    bot_token = os.getenv("TELEGRAM_BOT_TOKEN")
    chat_id = os.getenv("TELEGRAM_CHAT_ID")
    texto = (
        "ğŸ“£ Curso Java 2025 disponible en Amazon Kindle ğŸ‡ªğŸ‡¸\n"

# publisher_mastodon.py
import os
import requests
from dotenv import load_dotenv

def publicar_en_mastodon():
    load_dotenv(".env_template_redes")
    token = os.getenv("MASTODON_ACCESS_TOKEN")
    base_url = os.getenv("MASTODON_API_BASE_URL")
    texto = (
        "ğŸ˜ 'Java Course 2025' ya en Kindle. Aprende Java como un profesional.\n"
        + os.getenv("BOOK_URL_ES") + "\n"
        + os.getenv("JAVA_IMAGE_URL")
    )
    response = requests.post(
        f"{base_url}/api/v1/statuses",
        headers={"Authorization": f"Bearer {token}"},
        data={"status": texto}
    )
    print("âœ… Publicado en Mastodon." if response.ok else f"âŒ Error: {response.text}")
